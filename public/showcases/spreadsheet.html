Spreadsheet

Excel-like grid with cell editing, formulas, sorting, and copy/paste

javascript 
const sheet = document.querySelector(
  'snice-spreadsheet');
sheet.columns = [
  { header: 'Product', type: 'text' },
  { header: 'Q1', type: 'number', width: 90 },
  { header: 'Q2', type: 'number', width: 90 },
  { header: 'Total', type: 'number', width: 90 }
];
sheet.data = [
  ['Widget A', 1200, 1450, '=SUM(B1:C1)'],
  ['Widget B', 800, 920, '=SUM(B2:C2)'],
  ['Total', '=SUM(B1:B2)', '=SUM(C1:C2)',
   '=SUM(D1:D2)']
];

// Formulas: SUM, AVG, COUNT, MIN, MAX
// Keyboard: arrows, Enter, Tab, Ctrl+C/V
