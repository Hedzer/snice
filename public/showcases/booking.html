Booking

Appointment booking widget with date picker, time slot selection, and confirmation form

javascript <snice-booking id="bk" duration="30" variant="stepper" ></snice-booking> <script> // Generate available dates const dates = []; for (let i = 1; i <= 14; i++) { const d = new Date(); d.setDate(d.getDate() + i); if (d.getDay() !== 0 && d.getDay() !== 6) dates.push(d.toISOString().slice(0,10)); } bk.availableDates = dates; bk.availableSlots = dates.flatMap( date => ['09:00','10:00','11:00', '14:00','15:00','16:00'].map( time => ({ date, time, duration: 30 }) )); bk.fields = [ { name: 'name', label: 'Full Name', type: 'text', required: true }, { name: 'email', label: 'Email', type: 'email', required: true }, { name: 'notes', label: 'Notes', type: 'textarea' } ]; bk.addEventListener('booking-confirm', e => console.log(e.detail)); </script>
