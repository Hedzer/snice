Infinite Scroll

Container that loads more content as the user scrolls near the bottom
html 
<snice-infinite-scroll
  threshold="200"
  has-more
>
  <!-- scrollable content -->
</snice-infinite-scroll>

<script>
  const scroller = document.querySelector(
    'snice-infinite-scroll');
  scroller.addEventListener('load-more',
    async () => {
    scroller.loading = true;
    const items = await fetchMore();
    appendItems(items);
    scroller.loading = false;
    if (!items.length)
      scroller.hasMore = false;
  });
</script>
