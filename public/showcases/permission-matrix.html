Permission Matrix

Role-permission grid with toggleable checkboxes, grouped permissions, and read-only mode

javascript <snice-permission-matrix id="pm" ></snice-permission-matrix> <script> pm.roles = [ { id: 'admin', name: 'Admin' }, { id: 'editor', name: 'Editor' }, { id: 'viewer', name: 'Viewer' } ]; pm.permissions = [ { id: 'read', name: 'Read', group: 'Content' }, { id: 'write', name: 'Write', group: 'Content' }, { id: 'delete', name: 'Delete', group: 'Content' }, { id: 'manage', name: 'Manage Users', group: 'Admin' } ]; pm.matrix = { admin: ['read','write','delete','manage'], editor: ['read', 'write'], viewer: ['read'] }; pm.addEventListener('permission-toggle', e => console.log(e.detail)); </script>
