Comments

Threaded comment section with nested replies, likes, avatars, and timestamps

html 
<snice-comments
  current-user="Alice"
></snice-comments>

<script>
  const el = document.querySelector(
    'snice-comments');
  el.comments = [
    { id: '1', author: 'Bob',
      avatar: 'avatar.jpg',
      text: 'Great post!',
      timestamp: new Date().toISOString(),
      likes: 3, liked: false,
      replies: [
        { id: '2', author: 'Alice',
          text: 'Thanks!',
          timestamp: new Date().toISOString()
        }
      ]
    }
  ];
  el.addEventListener('comment-add',
    e => console.log(e.detail));
  el.addEventListener('comment-like',
    e => console.log(e.detail));
</script>
