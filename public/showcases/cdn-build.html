CDN Build

Use any component without the full framework. Load the runtime once, then add components as needed.

npx snice build-component button
# creates snice-button.min.js (~2-5kb)

<!-- Load runtime once -->
<script src="snice-runtime.min.js"></script>
<!-- Add components -->
<script src="snice-button.min.js"></script>
<snice-button variant="primary">Works anywhere</snice-button>
